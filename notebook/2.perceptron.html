class Perceptron_self(object):
    def __init__(self):
        self.w = np.ones(df.shape[1] - 1, dtype=np.float32)
        self.b = 0
        self.eta = 0.1
        
    def sign(self, x, w, b):
        y = np.dot(x,w) + b
        return y
    
    def fit(self, X, Y):
        while True:
            count = 0
            for i in range(len(X)):
                x = X[i]
                y = Y[i]
                if y * self.sign(x, self.w, self.b) <= 0:
                    self.w = self.w + self.eta * np.dot(x,y)
                    self.b = self.b + self.eta * y
                    count += 1
            if count == 0:
                return 'Finish'
